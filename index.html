private static int[] NaiveStringSearch(string text, 
               string pattern, bool overlap)
{
    int[] result = Array.Empty<int>();
    for (int i = 0; i < text.Length - pattern.Length + 1; i++)
    {
        var a = text[i.. (i + pattern.Length)];
        if (a == pattern)
        {
            result = AddElementToArray(result, i + 1);
            if (!overlap)
            {
                i += pattern.Length - 1;
            }
        }
    }
    return result;
}
